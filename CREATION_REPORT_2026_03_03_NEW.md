## 🚀 使用示例

### 使用渐变背景

```tsx
import CyberGradientBackground from '@/components/effects/CyberGradientBackground';

<CyberGradientBackground
  colors={['#00f0ff', '#9d00ff', '#ff0080']}
  speed={1}
  intensity={0.5}
>
  <YourContent />
</CyberGradientBackground>
```

### 使用粒子效果

```tsx
import CyberParticles from '@/components/effects/CyberParticles';

<CyberParticles
  particleCount={100}
  color="#00f0ff"
  connectionDistance={150}
  mouseInteraction={true}
/>
```

### 使用表单组件

```tsx
import { CyberInput, CyberSelect, CyberButton } from '@/components/ui';

<CyberInput
  label="用户名"
  variant="glow"
  error={errors.username}
  helperText="请输入您的用户名"
/>

<CyberSelect
  label="选择主题"
  variant="neon"
  options={[
    { value: 'dark', label: '深色主题' },
    { value: 'light', label: '浅色主题' },
  ]}
/>
```

### 使用工具函数

```tsx
import { formatCurrency, formatRelativeTime, getCyberGradient } from '@/lib/utils';

// 格式化货币
const price = formatCurrency(99.99, 'USD'); // "$99.99"

// 格式化相对时间
const time = formatRelativeTime(new Date()); // "2 hours ago"

// 获取赛博朋克渐变
const gradient = getCyberGradient('cyan', 'purple');
```
