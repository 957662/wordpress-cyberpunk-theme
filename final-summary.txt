【前端 - 使用 WordPress Hooks】

```typescript
'use client';

import { usePosts } from '@/lib/wordpress/hooks-complete';

export default function BlogPage() {
  const { data: posts, isLoading } = usePosts({
    page: 1,
    perPage: 10,
    orderBy: 'date',
  });

  if (isLoading) return <div>Loading...</div>;
  return <div>{posts?.map(p => <div key={p.id}>{p.title.rendered}</div>)}</div>;
}
```

【前端 - 使用博客组件】

```typescript
import { BlogList } from '@/components/blog/BlogListFixed';

export default function Page() {
  return <BlogList posts={posts} viewMode="grid" />;
}
```

【后端 - API 调用】

```bash
# 获取文章列表
GET /api/v1/blog/posts?page=1&per_page=10

# 获取文章详情
GET /api/v1/blog/posts/123

# 创建文章
POST /api/v1/blog/posts
```
