### 示例代码
```typescript
function BlogPage() {
  const { data: posts, isLoading } = usePosts({ page: 1, pageSize: 12 });

  if (isLoading) return <div>加载中...</div>;

  return <BlogGrid posts={posts} columns={3} />;
}
```
