## 📝 使用示例

### 导入组件

```tsx
// 导入单个组件
import { CyberButton } from '@/components/ui/CyberButton';

// 导入多个组件
import { HomeIcon, GitHubIcon, SearchIcon } from '@/components/graphics';

// 从索引导入
import * as Components from '@/components';
```

### 使用图形组件

```tsx
import { HomeIcon, MainLogo, CornerBracket } from '@/components/graphics';

// 图标
<HomeIcon size={24} className="text-cyber-cyan" />

// Logo
<MainLogo width={200} />

// 装饰
<CornerBracket position="top-left" />
```

### 使用UI组件

```tsx
import { CyberButton, Card, Input } from '@/components/ui';

// 按钮
<CyberButton variant="primary" size="lg">
  点击我
</CyberButton>

// 卡片
<Card className="p-6">
  <h3>标题</h3>
  <p>内容</p>
</Card>

// 输入框
<Input placeholder="请输入..." />
```

### 使用博客组件

```tsx
import { ArticleCard, BlogGrid } from '@/components/blog';

// 文章卡片
<ArticleCard article={articleData} />

// 博客网格
<BlogGrid limit={6} columns={3} />
```
