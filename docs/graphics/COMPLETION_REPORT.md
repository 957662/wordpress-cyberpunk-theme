## 💡 使用示例

### 基础使用
```tsx
import { Logo, Icon } from '@/components/ui';
import { NeonText, CyberCard } from '@/components/effects';

<Logo variant="horizontal" size={120} />
<Icon name="home" size={24} color="cyan" />
<NeonText color="cyan">霓虹文字</NeonText>
<CyberCard variant="neon">卡片</CyberCard>
```

### 高级使用
```tsx
<CyberCard variant="holographic" color="cyan" hover>
  <div className="flex gap-2">
    <CyberBadge variant="solid" color="cyan">标签</CyberBadge>
  </div>
  <h3><NeonText color="cyan">标题</NeonText></h3>
  <p>内容</p>
</CyberCard>
```
