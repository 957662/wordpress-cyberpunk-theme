收藏 API:
  GET    /api/bookmarks              → 获取收藏列表
  POST   /api/bookmarks              → 添加收藏
  DELETE /api/bookmarks/:id          → 删除收藏

收藏夹 API:
  GET    /api/bookmarks/folders      → 获取收藏夹列表
  POST   /api/bookmarks/folders      → 创建收藏夹
  PATCH  /api/bookmarks/folders      → 更新收藏夹
  DELETE /api/bookmarks/folders/:id  → 删除收藏夹

通知 API:
  GET    /api/notifications          → 获取通知列表
  POST   /api/notifications          → 批量操作
  PATCH  /api/notifications          → 更新通知
  DELETE /api/notifications          → 删除通知

// 使用 Hook
import { useBookmarks } from '@/hooks/useBookmarks';

function MyComponent() {
  const { bookmarks, toggleBookmark } = useBookmarks();
  
  return (
    <div>
      {bookmarks.map(item => (
        <BookmarkCard 
          key={item.id} 
          item={item}
          onToggle={() => toggleBookmark(item.targetId, 'post')}
        />
      ))}
    </div>
  );
}
