## 💡 提示与技巧

### 性能优化
```tsx
// ✅ 按需导入
import { SearchIcon } from '@/components/icons/SearchIcon';

// ❌ 避免全量导入
import * as Icons from '@/components/icons';
```

### 可访问性
```tsx
// ✅ 添加 aria-label
<button aria-label="搜索">
  <SearchIcon size={20} />
</button>

// ✅ 装饰性图标
<div aria-hidden="true">
  <CpuIcon size={64} />
</div>
```

### 响应式尺寸
```tsx
// 根据断点调整尺寸
<SearchIcon
  size={24}
  className="w-6 h-6 sm:w-8 sm:h-8 lg:w-10 lg:h-10"
/>
```

### 动画使用
```tsx
// ✅ 适度使用
<LoadingIcon animated={true} />
<SyncIcon animated={true} />
<OnlineIcon animated={true} />

// ❌ 避免过度使用
<HomeIcon animated={true} />  // 不必要
```
