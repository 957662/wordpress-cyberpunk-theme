## 📐 图标尺寸标准

```css
/* 极小 */
.icon-xs { width: 12px; height: 12px; }

.icon-sm { width: 16px; height: 16px; }

.icon-md { width: 24px; height: 24px; }
```

## 🔧 图标使用示例

### 基础使用
```tsx
import { SearchIcon, SettingsIcon, UserIcon } from '@/components/icons';

function Header() {
  return (
    <div className="flex items-center gap-4">
      <SearchIcon size={24} className="text-cyber-cyan" />
      <SettingsIcon size={24} className="text-cyber-purple" />
      <UserIcon size={24} className="text-cyber-pink" />
    </div>
  );
}
```

### 带发光效果
```tsx
function GlowingButton() {
  return (
    <button className="flex items-center gap-2 px-4 py-2 bg-cyber-cyan/10 border border-cyber-cyan rounded-lg hover:bg-cyber-cyan/20 transition-all">
      <SearchIcon className="text-cyber-cyan drop-shadow-[0_0_8px_rgba(0,240,255,0.8)]" />
      <span className="text-cyber-cyan">搜索</span>
    </button>
  );
}
```

### 动态颜色
```tsx
function ThemedIcon({ theme }: { theme: 'cyan' | 'purple' | 'pink' }) {
  const colors = {
    cyan: 'text-cyber-cyan',
    purple: 'text-cyber-purple',
    pink: 'text-cyber-pink',
  };

  return (
    <StarIcon className={colors[theme]} />
  );
}
```
